    avue-curd表格常用属性和用法

    JinJuntong 7/23/2021 avuevue

    个人用法：在 vue 中可以单独建立一个 option.js 文件，放置默认的统一参数，在使用的 vue 组件中引入该文件，设置组件内特殊的参数。

    在单独 vue 中间引入并使用

    <avue-crud
  v-if="option.column.length && redrawList"
    :option="option"
    :table-loading="option.loading"
    :data="data"
    :page.sync="page"	//页面参数
    @on-load="onLoad" 	//加载 用于初始数据获取 更新数据等
	@row-save="rowSave" // 行新增
	@row-del="rowDel" // 行删除
	@row-update="rowUpdate" // 行编辑
    @cell-click="cellClick"  //单元格点击方法(row,column,cell,event), 一般可以用来获取该行的ID
    @current-change="currentChange"	//当前页改变
    @size-change="sizeChange"	//每页数量改变
    @refresh-change="refreshChange"	//刷新按钮触发
    @search-change="searchChange"	//表格搜索事件
	>
</avue-crud>

import { cardOption } from 'options'
const option = cardOption({
  selection: true,
  menu: true,	// 菜单栏
  addBtn: false,
  editBtn: false,
  menuWidth: 220,
  delBtn: true,
  viewFormBtn: true,
  column:[]
})

export default{
	data() {
		return {
			option,
			data,
			page: {
		        pageSize: 20,
		        currentPage: 1,
		        total: 0，
		        search:{}
		    },
		}
	},
	methods:{
		// 点击刷新按钮
		refreshChange() {
	      this.onLoad(this.page, this.search);
	    },
 	    // 当前页码改变
	    currentChange(currentPage) {
	      this.page.currentPage = currentPage;
	    },
	    // 每页数量改变
	    sizeChange(pageSize) {
	      this.page.pageSize = pageSize;
	    },
	   	// 表格搜索
	    searchChange(params, done) {
	      this.page.currentPage = 1;
	      this.search = params;
	      this.getThreeELementsList(this.page, this.search);
	      done();
	    },
	     // 列表
	    onLoad(page, params = {}) {
	      this.loading = true;
	      threeElementsList({
	        pageNum: page.currentPage,
	        pageSize: page.pageSize,
	        ...Object.assign(params, this.search)
	      })
	        .then(res => {
	          const { code, data } = res;
	          if (code === 200) {
	            const list = data.list;
	            this.curdData = list;
	            this.page.total = data.total;
	          }
	        })
	        .finally(() => {
	          this.loading = false;
	        });
	    },
		rowSave(row, done) {
			addPersonVillage(row.id).then(() => {
				done();
				this.onLoad();
				this.$message({
				type: "success",
				message: "操作成功!",
				});
			});
		},
		rowDel(row) {
			this.$confirm("确定将选择数据删除?", {
				confirmButtonText: "确定",
				cancelButtonText: "取消",
				type: "warning",
			})
				.then(() => {
				return delPersonVillage(row.id);
				})
				.then(() => {
				this.onLoad();
				this.$message({
					type: "success",
					message: "操作成功!",
				});
			});
		},
		rowUpdate(row, index, done, loading) {
			addPersonVillage(row).then(
				() => {
				done();
				this.onLoad(this.page);
				this.$message({
					type: "success",
					message: "操作成功!",
				});
				},
				(error) => {
				window.console.log(error);
				loading();
				}
			);
		},		
	}
}
