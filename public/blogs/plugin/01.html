    nprogress 进度条插件

    JinJuntong 7/29/2021 plugin

    nprogress官网 (opens new window)

    nprogress.js是一款进度条插件，个人用于vue的router.beforeEach()中。

    安装方法

    # 1、npm 安装

    npm install --save nprogress   (--save:开发也需要用到的依赖)
    1

    # 2、CDN引入

    <script src='nprogress.js'></script>
<link rel='stylesheet' href='nprogress.css'/>
    1
    2

    api

    NProgress.start();
NProgress.done();
NProgress.set(0.0);     // Sorta same as .start()
NProgress.set(0.4);
NProgress.set(1.0);     // Sorta same as .done()
NProgress.inc(); 		// 随机增长，永远不会到1

    实际使用

    //permission.js

import NProgress from 'nprogress' // progress bar
import 'nprogress/nprogress.css' // progress bar style

NProgress.configure({showSpinner: false}); 	// 进度环隐藏

router.beforeEach((to, from, next) => {
	NProgress.start();
	...
	...
	...
	NPogress.done()
})

router.afterEach(() => {
  NProgress.done()
})
