    path API

    JinJuntong 2/26/2022 node

    # path.join([...paths])

      用于将path片段连接在一起并规范化， path必须为字符串。

    🌰

    const path = requrire('path')
fn(path) {
    return path.join(__dirname, path)
}
fn('.file/01.text')
//当前文件目录下\file\01.text
//注意的是 ../ 会抵消前面的路径

path.join('/foo', 'bar', 'baz/asdf', 'quux', '..');
// 返回: '/foo/bar/baz/asdf'

path.join('foo', {}, 'bar');
// 抛出 'TypeError: Path must be a string. Received {}'

    # path.basename(path[, ext])

      返回path的最后一部分，一般为文件名字。 第二个参数为文件扩展名，可选参数。

    🌰

    path.basename(__dirname);
path.basename("c://目录1//目录2/文件名.txt"); //文件名.txt
path.basename("c://目录1//目录2/文件名.txt", ".txt"); //文件名

    # path.dirname(path)

      返回path的目录名字。 必须为字符串

    🌰

    path.dirname("/foo/bar/baz/asdf/quux");
// 返回: '/foo/bar/baz/asdf'

    # path.parse(path)

      返回一个对象，包含path的重要元素。

    🌰

    path.parse("C:\\path\\dir\\file.txt");

// 返回:
// { root: 'C:\\',
//   dir: 'C:\\path\\dir',
//   base: 'file.txt',
//   ext: '.txt',
//   name: 'file' }
